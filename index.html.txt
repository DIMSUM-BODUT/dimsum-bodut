        // Download CSV
        function downloadCSV(filename, data) {
            const csvContent = "text/csv;charset=utf-8," + encodeURIComponent(data);
            const link = document.createElement("a");
            link.setAttribute("href", csvContent);
            link.setAttribute("download", filename);
            link.style.visibility = "hidden";
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }

        // Download orders CSV
        function downloadOrdersCSV(ordersToDownload, filenamePrefix) {
            if (ordersToDownload.length === 0) {
                alert('Tidak ada data pesanan untuk diunduh!');
                return;
            }
            
            const headers = ['No Order', 'Nama Pemesan', 'Produk', 'Qty', 'Total', 'No HP', 'Alamat', 'Tanggal'];
            const csvContent = [
                headers.join(','),
                ...ordersToDownload.map(order => 
                    `"${order.orderNumber}","${order.customerName}","${order.productName}","${order.qty}","${order.totalPrice}","${order.phone}","${order.address}","${order.timestamp}"`
                )
            ].join('\n');
            
            const filename = `${filenamePrefix}_${new Date().toISOString().split('T')[0]}.csv`;
            downloadCSV(filename, csvContent);
        }
